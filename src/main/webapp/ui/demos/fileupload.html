1、上传图片带预览功能 ,多选

注意事项：

h5上传的一些限制：html代码上配置多选multiple,特定类型的上传文件配置accept

不支持h5的在uploadify属性配置，具体参考uploadify插件 

			//html代码
			<input type="file" name="files" id="fileupload" multiple accept="image/*">
			//js代码
			$('#fileupload').fileupload({
				//这个上传接口是bob提供
				url: '../../autogen/tools/upload.do',
				//data的参数也是bob接口要求
				data: {
					method: 'upload',
					appName: 'secure',
					dir: '/msoft/downloader/upload_test/'
				},
				//如果上传的是图片可以预览
				preview: true,
				text: '选择文件',
				//上传文件的大小限制
				size: '5MB',
				success: function(event,json){
					console.log(json.msg);
				},
				error: function(event,json){},
				//针对uploadify控件配置属性
				uploadify: {
					onUploadComplete: function(file){
						// alert(file.name);
					},
					//文件类型描述
					fileTypeDesc : 'Image Files',
					//文件类型限制
					fileTypeExts : '*.gif; *.jpg; *.png',
					width: 90,
					height: 25
				}
			});
