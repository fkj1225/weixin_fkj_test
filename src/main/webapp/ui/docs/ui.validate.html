Example

$('form').validate({
	regexp: {
		idcard: [/^\d{17}[\dxX]$/, '请输入正确的身份证号码'],
		fax: [/^0\d{2,3}\-\d{7,8}$/,'请输入正确的传真，格式：020-11112222', true] //true表示可以为空
		chk: [isChkOne, '最少选择一项']
	}
});
	//定义一个方法，替代正则，可以做一些不好用正则去做验证的事情，返回true表示验证通过
function isChkOne(){
	//code
}


注：这个方法适合批量处理表单验证，不能直接使用单个表单调用，比如：$('input').validate();
一定要使用表单的包裹层去调用
